● 下面给出在 Windows 上完整搭建的步骤和所有配置文件。

  ---
  整体架构

  VSCode (编辑/调试)
    └─ CMake + MinGW-w64 (构建系统)
         └─ main.c (你的 LVGL 应用代码)
              ├─ lvgl (LVGL 核心库)
              └─ SDL2 (显示 + 鼠标/键盘输入)

  ---
  第一步：安装工具链

  1.1 安装 MSYS2

  从 https://www.msys2.org/ 下载安装。装完后打开 MSYS2 UCRT64 终端，执行：

  pacman -Syu
  # 重启终端后再执行：
  pacman -S mingw-w64-ucrt-x86_64-gcc \
             mingw-w64-ucrt-x86_64-cmake \
             mingw-w64-ucrt-x86_64-ninja \
             mingw-w64-ucrt-x86_64-gdb \
             mingw-w64-ucrt-x86_64-SDL2 \
             git

  1.2 配置环境变量

  将以下路径加入系统 PATH：

  C:\msys64\ucrt64\bin

  验证：

  gcc --version
  cmake --version
  sdl2-config --cflags --libs

  ---
  第二步：创建项目结构

  lvgl_sim/
  ├── CMakeLists.txt
  ├── .vscode/
  │   ├── c_cpp_properties.json
  │   ├── tasks.json
  │   └── launch.json
  ├── lv_conf.h
  └── main/
      └── main.c

  2.1 克隆 LVGL

  cd lvgl_sim
  git clone https://github.com/lvgl/lvgl.git --depth 1

  ---
  第三步：配置文件

  3.1 CMakeLists.txt（项目根目录）

  cmake_minimum_required(VERSION 3.16)
  project(lvgl_sim C)

  set(CMAKE_C_STANDARD 99)
  set(CMAKE_C_STANDARD_REQUIRED ON)

  # ── SDL2 ──────────────────────────────────
  find_package(SDL2 REQUIRED)
  include_directories(${SDL2_INCLUDE_DIRS})

  # ── LVGL ──────────────────────────────────
  set(LVGL_DIR ${CMAKE_CURRENT_SOURCE_DIR}/lvgl)
  add_subdirectory(${LVGL_DIR} ${CMAKE_BINARY_DIR}/lvgl)
  include_directories(
      ${CMAKE_CURRENT_SOURCE_DIR}          # lv_conf.h 所在目录
      ${LVGL_DIR}
      ${LVGL_DIR}/src
  )

  # ── 应用源码 ──────────────────────────────
  file(GLOB SOURCES main/*.c)

  add_executable(${PROJECT_NAME}
      ${SOURCES}
      lvgl/examples/porting/lv_port_disp_template.c
      lvgl/examples/porting/lv_port_indev_template.c
  )

  target_link_libraries(${PROJECT_NAME}
      PRIVATE lvgl
      PRIVATE ${SDL2_LIBRARIES}
  )

  3.2 lv_conf.h（项目根目录）

  从 lvgl/lv_conf_template.h 复制一份并改名为 lv_conf.h，然后修改以下关键项：

  // 必须取消这行注释
  #if 1   // ← 改为 1

  #define LV_COLOR_DEPTH 32          // ARGB8888，配合 SDL2
  #define LV_MEM_SIZE (1024 * 1024)  // 1 MB 堆内存

  #define LV_USE_LOG 1               // 开启日志，调试用

  // 开启需要用到的控件/功能
  #define LV_USE_BTN      1
  #define LV_USE_LABEL    1
  #define LV_USE_SLIDER   1
  #define LV_USE_CHART    1
  #define LV_USE_DEMO_WIDGETS  1    // 可先用官方案例验证

  // 帧率与性能
  #define LV_DISP_DEF_REFR_PERIOD 30   // ~33 FPS

  3.3 main/main.c

  #include "lvgl/lvgl.h"
  #include "lvgl/examples/lv_examples.h"

  #if defined(_WIN32)
  #include <windows.h>
  #endif

  // SDL 头文件
  #include <SDL2/SDL.h>

  /* ── 显示分辨率（虚拟屏幕） ── */
  #define HOR_RES  480
  #define VER_RES  320

  /* ── 全局变量 ── */
  static SDL_Window   *window   = NULL;
  static SDL_Renderer *renderer = NULL;
  static SDL_Texture  *texture  = NULL;
  static lv_disp_draw_buf_t draw_buf;
  static lv_color_t buf1[HOR_RES * VER_RES];

  /* ── 显示刷新回调 ── */
  static void disp_flush(lv_disp_drv_t *disp_drv, const lv_area_t *area,
                         lv_color_t *color_p)
  {
      int w = area->x2 - area->x1 + 1;
      int h = area->y2 - area->y1 + 1;

      SDL_Rect rect = { area->x1, area->y1, w, h };
      SDL_UpdateTexture(texture, &rect, color_p, w * sizeof(lv_color_t));
      SDL_RenderCopy(renderer, &rect, &rect);

      lv_disp_flush_ready(disp_drv);
  }

  /* ── 输入读取回调（触摸 + 编码器按键） ── */
  static void touchpad_read(lv_indev_drv_t *indev_drv, lv_indev_data_t *data)
  {
      static lv_coord_t last_x = 0, last_y = 0;
      int x, y;
      uint32_t btn = SDL_GetMouseState(&x, &y);

      if (btn & SDL_BUTTON(SDL_BUTTON_LEFT)) {
          data->point.x = x;
          data->point.y = y;
          last_x = x;
          last_y = y;
          data->state = LV_INDEV_STATE_PR;
      } else {
          data->point.x = last_x;
          data->point.y = last_y;
          data->state = LV_INDEV_STATE_REL;
      }
  }

  static int main(void)
  {
      /* ── SDL 初始化 ── */
      SDL_Init(SDL_INIT_VIDEO);
      window = SDL_CreateWindow("LVGL Simulator",
                                SDL_WINDOWPOS_CENTERED,
                                SDL_WINDOWPOS_CENTERED,
                                HOR_RES, VER_RES, 0);
      renderer = SDL_CreateRenderer(window, -1, SDL_RENDERER_ACCELERATED);
      texture  = SDL_CreateTexture(renderer,
                                   SDL_PIXELFORMAT_ARGB8888,
                                   SDL_TEXTUREACCESS_STREAMING,
                                   HOR_RES, VER_RES);

      /* ── LVGL 初始化 ── */
      lv_init();
      lv_disp_draw_buf_init(&draw_buf, buf1, NULL, HOR_RES * VER_RES);

      static lv_disp_drv_t disp_drv;
      lv_disp_drv_init(&disp_drv);
      disp_drv.hor_res  = HOR_RES;
      disp_drv.ver_res  = VER_RES;
      disp_drv.flush_cb = disp_flush;
      disp_drv.draw_buf = &draw_buf;
      lv_disp_drv_register(&disp_drv);

      static lv_indev_drv_t indev_drv;
      lv_indev_drv_init(&indev_drv);
      indev_drv.type      = LV_INDEV_TYPE_POINTER;
      indev_drv.read_cb   = touchpad_read;
      lv_indev_drv_register(&indev_drv);

      /* ── 你的 GUI 代码 ── */
      lv_demo_widgets();   // 先用官方组件 demo 验证环境

      /* ── 主循环 ── */
      int running = 1;
      while (running) {
          lv_timer_handler();
          SDL_Delay(5);

          SDL_RenderPresent(renderer);

          SDL_Event e;
          while (SDL_PollEvent(&e)) {
              if (e.type == SDL_QUIT) running = 0;
          }
      }

      SDL_DestroyTexture(texture);
      SDL_DestroyRenderer(renderer);
      SDL_DestroyWindow(window);
      SDL_Quit();
      return 0;
  }

  ---
  第四步：编译 & 运行

  4.1 命令行构建

  在 MSYS2 UCRT64 终端中（确保终端 PATH 里有 cmake 和 ninja）：

  cd lvgl_sim
  cmake -G Ninja -DCMAKE_BUILD_TYPE=Debug -B build
  cmake --build build
  ./build/lvgl_sim.exe

  4.2 VSCode 一键调试

  1. 用 VSCode 打开 lvgl_sim 文件夹
  2. 安装扩展：C/C++ (Microsoft)、CMake Tools (Microsoft)
  3. 按 Ctrl+Shift+B 编译
  4. 按 F5 启动调试 — 窗口弹出，LVGL Demo 运行

  ---
  第五步：快速迁移到嵌入式

  模拟器和实际嵌入式的 GUI 代码可以 完全共用，只需替换三个部分：
  层次: 显示驱动
  PC 模拟器: disp_flush → SDL2 Texture
  嵌入式（STM32 + TFT）: disp_flush → 写 ILI9341 显存
  ────────────────────────────────────────
  层次: 输入驱动
  PC 模拟器: touchpad_read → SDL 鼠标
  嵌入式（STM32 + TFT）: touchpad_read → XPT2046 触摸 IC
  ────────────────────────────────────────
  层次: tick
  PC 模拟器: SDL_Delay(5) 即可
  嵌入式（STM32 + TFT）: 硬件定时器 1ms 中断中调用 lv_tick_inc(1)
  你的 main.c 中 /* 你的 GUI 代码 */ 那部分直接复制到嵌入式工程，LVGL 核心库和
  lv_conf.h 也同步过去即可。

  ---
  常见问题

  Q: CMake 报 Could NOT find SDL2？
  A: 确认 sdl2-config --cflags --libs 有输出。如果不行，在 CMakeLists.txt 
  里用绝对路径：
  set(SDL2_INCLUDE_DIRS "C:/msys64/ucrt64/include/SDL2")
  set(SDL2_LIBRARIES    "-L" "C:/msys64/ucrt64/lib" "-lSDL2main" "-lSDL2" 
  "-lmingw32")

  Q: 运行窗口闪退？
  A: 检查 lv_conf.h 中的 #if 0 是否改为 #if 1，以及 LV_COLOR_DEPTH 是否设为 32。

  Q: SDL_UpdateTexture 闪烁？
  A: 可以用双缓冲（再声明一个 buf2，lv_disp_draw_buf_init(&draw_buf, buf1, buf2,
   ...)），或者改用 disp_flush 回调只标记脏区、在 lv_timer_handler 后统一
  SDL_RenderPresent。
